02

useQuery — Fetching Data

Cara paling dasar ambil data dari API
typescript
// src/hooks/useUsers.ts
import { useQuery } from '@tanstack/react-query'

type User = {
  id: number
  name: string
  email: string
}

export function useUsers() {
  return useQuery<User[]>({
    queryKey: ['users'],
    queryFn: async () => {
      const res = await fetch('https://jsonplaceholder.typicode.com/users')
      if (!res.ok) throw new Error('Gagal fetch users')
      return res.json()
    },
  })
}

// Penggunaan di komponen
function UserList() {
  const { data, isLoading, isError, error } = useUsers()

  if (isLoading) return <div>Loading...</div>
  if (isError) return <div>Error: {error.message}</div>

  return <ul>{data?.map(u => <li key={u.id}>{u.name}</li>)}</ul>
}
03

useMutation — POST / PUT / DELETE

Mutation dengan optimistic update
typescript
import { useMutation, useQueryClient } from '@tanstack/react-query'

export function useCreateUser() {
  const qc = useQueryClient()

  return useMutation({
    mutationFn: (newUser: Omit<User, 'id'>) =>
      fetch('/api/users', {
        method: 'POST',
        body: JSON.stringify(newUser),
        headers: { 'Content-Type': 'application/json' },
      }).then(r => r.json()),

    // Invalidate cache setelah sukses → refetch otomatis!
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['users'] })
    },
  })
}
✅

Pro tip: Gunakan invalidateQueries untuk auto-refetch data setelah mutation. Ini bikin UI selalu sinkron dengan server tanpa manual setState.

02

Dynamic Route & Route Loader

Params + data loading sebelum render
typescript
// src/routes/users/$id.tsx
import { createFileRoute } from '@tanstack/react-router'

export const Route = createFileRoute('/users/$id')({
  // Loader: fetch data SEBELUM komponen render
  loader: async ({ params }) => {
    const user = await fetch(`/api/users/${params.id}`).then(r => r.json())
    return { user }
  },

  component: function UserDetail() {
    const { user } = Route.useLoaderData() // fully typed!
    const { id } = Route.useParams()       // typed params

    return <div>{user.name}</div>
  },
})
✅

Game changer: Route loader terintegrasi sempurna dengan TanStack Query. Kamu bisa preload query di loader sehingga data sudah ada saat komponen mount — tidak ada loading state!

01

Basic Table Setup

Column definition + useReactTable
typescript
import {
  useReactTable, getCoreRowModel,
  getSortedRowModel, getFilteredRowModel,
  flexRender, ColumnDef
} from '@tanstack/react-table'

const columns: ColumnDef<User>[] = [
  { accessorKey: 'id', header: 'ID' },
  { accessorKey: 'name', header: 'Nama' },
  { accessorKey: 'email', header: 'Email' },
  {
    id: 'actions',
    header: 'Aksi',
    cell: ({ row }) => (
      <button onClick={() => handleEdit(row.original.id)}>
        Edit
      </button>
    ),
  },
]

function UserTable({ data }: { data: User[] }) {
  const table = useReactTable({
    data,
    columns,
    getCoreRowModel: getCoreRowModel(),
    getSortedRowModel: getSortedRowModel(),  // sorting
    getFilteredRowModel: getFilteredRowModel(), // filter
  })

  return (
    <table>
      <thead>{table.getHeaderGroups().map(hg => (
        <tr key={hg.id}>{hg.headers.map(h => (
          <th key={h.id}>
            {flexRender(h.column.columnDef.header, h.getContext())}
          </th>
        ))}</tr>
      ))}</thead>
      <tbody>{table.getRowModel().rows.map(row => (
        <tr key={row.id}>{row.getVisibleCells().map(cell => (
          <td key={cell.id}>
            {flexRender(cell.column.columnDef.cell, cell.getContext())}
          </td>
        ))}</tr>
      ))}</tbody>
    </table>
  )
}

Saran Developer: TanStack Form masih relatif baru tapi sangat powerful. Keunggulannya ada di validasi yang benar-benar type-safe dan async validation bawaan. Kombinasikan dengan Zod untuk skema validasi yang lengkap.

01

Basic Form dengan Zod Validation

useForm + field validation
bash
# Install Zod untuk validasi schema
pnpm add zod @tanstack/zod-form-adapter
typescript
import { useForm } from '@tanstack/react-form'
import { zodValidator } from '@tanstack/zod-form-adapter'
import { z } from 'zod'

const schema = z.object({
  name: z.string().min(3, 'Nama minimal 3 karakter'),
  email: z.string().email('Email tidak valid'),
})

function UserForm() {
  const form = useForm({
    defaultValues: { name: '', email: '' },
    validatorAdapter: zodValidator(),
    validators: { onChange: schema },
    onSubmit: async ({ value }) => {
      console.log('Submit:', value)
    },
  })

  return (
    <form onSubmit={e => { e.preventDefault(); form.handleSubmit() }}>
      <form.Field name="name"
        children={(field) => (
          <>
            <input
              value={field.state.value}
              onChange={e => field.handleChange(e.target.value)}
            />
            {field.state.meta.errors[0] && (
              <span style={{ color: 'red' }}>
                {field.state.meta.errors[0]}
              </span>
            )}
          </>
        )}
      />
      <button type="submit">Kirim</button>
    </form>
  )
}

01

Buat Store Global

Reactive state tanpa boilerplate
typescript
// src/stores/uiStore.ts
import { Store, useStore } from '@tanstack/react-store'

interface UIState {
  sidebarOpen: boolean
  theme: 'light' | 'dark'
  notifications: number
}

export const uiStore = new Store<UIState>({
  sidebarOpen: false,
  theme: 'dark',
  notifications: 0,
})

// Actions
export const toggleSidebar = () =>
  uiStore.setState(s => ({ ...s, sidebarOpen: !s.sidebarOpen }))

export const setTheme = (theme: UIState['theme']) =>
  uiStore.setState(s => ({ ...s, theme }))

// Penggunaan di komponen
function Navbar() {
  const sidebarOpen = useStore(uiStore, s => s.sidebarOpen)
  const theme = useStore(uiStore, s => s.theme)

  return (
    <nav>
      <button onClick={toggleSidebar}>
        {sidebarOpen ? 'Tutup' : 'Buka'} Sidebar
      </button>
      <span>Theme: {theme}</span>
    </nav>
  )
}

04

Checklist Sebelum Deploy

Production readiness
  • Environment variables sudah dipindah ke .env.production
  • React Query devtools hanya aktif di development
  • Error boundaries sudah dipasang untuk catch runtime errors
  • Loading states & skeleton UI untuk semua async operations
  • Lighthouse score > 90 untuk performance
  • Bundle size dicek dengan vite-bundle-visualizer
  • Route code splitting aktif (TanStack Router otomatis)
  • CORS sudah dikonfigurasi di backend
  • API base URL sudah dari environment variable
